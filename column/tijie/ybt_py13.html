                                                              # 【题目描述】

                                                              利用for循环。计算输出1+2+3+...+n的和。

                                                              # 【输入】

                                                              输入n。

                                                              # 【输出】

                                                              如题述，之和。

                                                              # 【输入样例】

                                                              10

                                                              # 【输出样例】

                                                              55

                                                              # 代码

                                                              n = int(input())
total = 0
for i in range(1, n+1):
    total += i
print(total)
