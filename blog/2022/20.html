def fun3():
    print('序号      电影名      票价      张数')
    for i,j in enumerate(selected,start=1):
        print('{}      {}      {}      {}'.format(i,j[0],j[1],count[i-1]))
def fun4():
    while True:
        print('删除前')
        print('序号      电影名      票价      张数')
        for i, j in enumerate(selected, start=1):
            print('{}      {}      {}      {}'.format(i, j[0], j[1], count[i - 1]))
        dlt = int(input('请输入要删除的订单序号：'))
        del selected[dlt - 1]
        del count[dlt - 1]
        print('删除后')
        print('序号      电影名      票价      张数')
        for i, j in enumerate(selected, start=1):
            print('{}      {}      {}      {}'.format(i, j[0], j[1], count[i - 1]))
        re_ctn=int(input('是否继续删除？（0退出/其他数字继续删除）：'))
        if re_ctn==0:
            break
